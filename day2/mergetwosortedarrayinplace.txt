#include <bits/stdc++.h> 
vector<int> ninjaAndSortedArrays(vector<int>& arr1, vector<int>& arr2, int m, int n) {
    int i=0;
    int j=0;
    vector<int> result;
    while(i<m && j<n){ 
      if(arr1[i]==arr2[j])
         { 
          result.push_back(arr1[i]);
          result.push_back(arr2[j]);
            j++;
            i++;
        }
      else if(arr1[i]<arr2[j]){
          result.push_back(arr1[i++]);
      }
    else if(arr1[i]>arr2[j]){
        result.push_back(arr2[j++]);
        }
    }
    while(i<m){
       result.push_back(arr1[i++]);
    }
    while(j<n){
        result.push_back(arr2[j++]);
    }
    return result;
}
